\flushleft \shadebox{$\Gamma \vdash \delta: \Delta$}
\begin{smathpar}
   \inferrule*[right={
      \textnormal{$\Delta = [f : \tyFun{A_f}{B_f} \mid f \in F]$}
   }]
   {
      \Gamma \concat \Delta \vdash \exFun{\sigma_f}: \tyFun{A_f}{B_f}
      \quad
      (\forall f \in F)
   }
   {
      \Gamma \vdash [\exFun{f}{\sigma_f} \mid f \in F]: \Delta
   }
\end{smathpar}
\\[5mm]
\flushleft \shadebox{$\Gamma \vdash e: A$}
\begin{smathpar}
   \inferrule*
   {
      \Gamma \vdash \sigma : \tyFun{A}{B}
   }
   {
      \Gamma \vdash \exLambda{\sigma} : \tyFun{A}{B}
   }
   %
   \and
   %
   \inferrule*
   {
      \Gamma \vdash \singleton{\sigma}{e'} : \tyFun{A}{B}
      \\
      \Gamma \vdash e : A
   }
   {
      \Gamma \vdash \exLetStructured{\singleton{\sigma}{e}}{e'} : B
   }
   %
   \and
   %
   \inferrule*
   {
      \Gamma \vdash \delta: \Delta
      \\
      \Gamma \concat \Delta \vdash e : A
   }
   {
      \Gamma \vdash \exLetRecMutual{\delta}{e} : A
   }
\end{smathpar}
