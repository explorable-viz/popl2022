\flushleft \shadebox{$\Gamma \vdash \sigma: \tyFun{A}{B}$}
\begin{smathpar}
\inferrule*
{
   \cxtExtend{\Gamma}{x}{A} \vdash \kappa: B
}
{
   \Gamma \vdash \elimVar{x}{\kappa}: \tyFun{A}{B}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash \kappa: A
   \\
   \Gamma \vdash \kappa': A
}
{
   \Gamma \vdash \elimBool{\kappa}{\kappa'}: \tyFun{\tyBool}{A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash \sigma: \tyFun{A}{\tyFun{A'}{B}}
}
{
   \Gamma \vdash \elimProd{\sigma}: \tyFun{\tyProd{A}{A'}}{B}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash \kappa: A
   \\
   \Gamma \vdash \sigma: \tyFun{\tyInt}{\tyFun{\tyList{\tyInt}}{A}}
}
{
   \Gamma \vdash \elimList{\branchNil{\kappa}}{\branchCons{\sigma}}: \tyFun{\tyList{\tyInt}}{A}
}
\end{smathpar}
\caption{Typing rules for eliminators}
\end{figure}
