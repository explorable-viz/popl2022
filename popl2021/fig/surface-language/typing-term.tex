\flushleft \shadebox{$\Gamma \vdash t: A$}
\begin{smathpar}
\inferrule*
{
   \Gamma \vdash t : A
   \\
   \Gamma \vdash \overrightarrow{t} : \tyList{A}
}
{
   \Gamma \vdash t \concat \overrightarrow{t} : \tyList{A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash t: A
   \\
   \Gamma \vdash \sigma: \tyFun{A}{B}
}
{
   \Gamma \vdash \exMatch{t}{\sigma}: B
}
%
\and
%
\inferrule*
{
   \Gamma \vdash t: \tyBool
   \\
   \Gamma \vdash t': A
   \\
   \Gamma \vdash t^{\twoPrime}: A
}
{
   \Gamma \vdash \exIfThenElse{t}{t'}{t^{\twoPrime}}: A
}
%
\and
%
\inferrule*
{
   \strut
}
{
   \Gamma \vdash \sExNil : \kw{List A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash t : \kw{A}
   \\
   \Gamma \vdash \overrightarrow{t} : \kw{List A}
}
{
   \Gamma \vdash \sExCons{t}{\overrightarrow{t}} : \kw{List A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash t: \tyInt
   \\
   \Gamma \vdash t': \tyInt
}
{
   \Gamma \vdash \exListSeq{t}{t'}: \tyList{\tyInt}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash \overrightarrow{q} \dashv \Delta
   \\
   \Gamma \concat \Delta \vdash t: A
}
{
   \Gamma \vdash \exListComp{t}{\overrightarrow{q}}: \tyList{A}
}
\end{smathpar}

\end{figure}
