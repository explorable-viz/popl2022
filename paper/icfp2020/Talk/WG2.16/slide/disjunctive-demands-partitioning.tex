\begin{frame}[t,fragile]
\frametitle{Disjunctive demands}

\vspace{10pt}
More generally a demand $\sigma$ of type $A$ must specify how much is
needed of \emph{every possible value} of type $A$.

\vspace{10pt}
\pause

Represents a \textbf{partitioning} of $A$ with a ``laziness'' property:

\only<3>{
\[
\Unk\quad
\begin{cases}
\Highlight{\ttt{[]}} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ []}} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ $\Unk$}}
\end{cases}
\]
}
\only<4>{
\[
\ttt{($\Unk$, $\Unk$) $\Cons$ $\Unk$}\quad
\begin{cases}
  \ttt{[]} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ []}} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ $\Unk$}}
\end{cases}
\]
}
\only<5>{
\[
\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ $\Unk$}\quad
\begin{cases}
\ttt{[]} \\
\ttt{($\Unk$, $\Unk$) $\Cons$ []} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ $\Unk$}}
\end{cases}
\]
}
\onslide<6->{
\[
\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ []}\quad
\begin{cases}
\ttt{[]} \\
\ttt{($\Unk$, $\Unk$) $\Cons$ []} \\
\Highlight{\ttt{($\Unk$, $\Unk$) $\Cons$ ($\Unk$, $\Unk$) $\Cons$ $\Unk$}}
\end{cases}
\]
}

\onslide<7->{
\begin{itemize}
\item determining the partition only requires partial knowledge of
the value
\end{itemize}
}
\end{frame}
