\begin{frame}
\frametitle{Demand-indexed evaluation}
Single elimination form, for any type $A$:
\begin{mathpar}
\inferrule*
{
   \Gamma \vdash e : A
   \\
   \Gamma \vdash \Trie{\sigma}{\textsf{Term}_B}{A}{}
}
{
   \Gamma \vdash \exMatch{e}{\sigma} : B
}
\end{mathpar}
Evaluation rule:
\begin{mathpar}
\inferrule*
{
  \Eeval{e}{\sigma}{u}{\theta}
  \\
  \Eeval{\substApp{\theta}{\sigma(u)}}{\tau}{v}{\theta'}
}
{
  \Eeval{\exMatch{e}{\sigma}}{\tau}{v}{\theta'}
}
\end{mathpar}
\end{frame}
