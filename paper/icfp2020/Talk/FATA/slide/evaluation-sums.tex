\begin{frame}
\frametitle{Demand-indexed evaluation}
Typing rule for a demand of type $\tySum{A}{B}$:
\begin{mathpar}
\inferrule*
{
  \Trie{\sigma}{X}{A}{}
  \\
  \Trie{\tau}{X}{B}{}
}
{
  \Trie{\trieSum{\sigma}{\tau}}{X}{(\tySum{A}{B})}{}
}
\end{mathpar}
Corresponding evaluation rules (intro only):
\begin{mathpar}
\inferrule*
{
  \Eeval{e}{\sigma}{v}{\theta}
}
{
  \Eeval{\exInl{e}}{\trieSum{\sigma}{\tau}}{\exInl{v}}{\theta}
}
%
\and
%
\inferrule*
{
  \Eeval{e}{\tau}{v}{\theta}
}
{
  \Eeval{\exInr{e}}{\trieSum{\sigma}{\tau}}{\exInr{v}}{\theta}
}
\end{mathpar}
\end{frame}
