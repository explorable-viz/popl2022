\flushleft \shadebox{$\Gamma \proj \Gamma'$}
\begin{smathpar}
   \inferrule*
   {
      \strut
   }
   {
      \cxtEmpty \proj \cxtEmpty
   }
   %
   \and
   %
   \inferrule*[right={$A' \neq \bot$}]
   {
      \Gamma \proj \Gamma'
      \\
      A \proj A'
   }
   {
      \cxtExtend{\Gamma}{x}{A} \proj \cxtExtend{\Gamma'}{x}{A'}
   }
   %
   \and
   %
   \inferrule*
   {
      \Gamma \proj \Gamma'
      \\
      A \proj \bot
   }
   {
      \cxtExtend{\Gamma}{x}{A} \proj \Gamma'
   }
\end{smathpar}
