\shadebox{$e \join e'$}
\begin{salign}
   \bot \join e &= e
   \\
   e \join \bot &= e
   \\
   \exVar{x} \join \exVar{x} &= \exVar{x}
   \\
   \exTrue \join \exTrue &= \exTrue
   \\
   \exFalse \join \exFalse &= \exFalse
   \\
   \exInt{n} \join \exInt{n} &= \exInt{n}
   \\
   \exPair{e_1}{e_2} \join \exPair{e_1'}{e_2'} &= \exPair{e_1 \join e_1'}{e_2 \join e_2'}
   \\
   \exPairProjL{e} \join \exPairProjL{e'} &= \exPairProjL{e \join e'}
   \\
   \exPairProjR{e} \join \exPairProjR{e'} &= \exPairProjR{e \join e'}
   \\
   \exPairProjL{e} \join \exPairProjR{e'} &= \exPair{e}{e'}
   \\
   \exPairProjR{e} \join \exPairProjL{e'} &= \exPair{e'}{e}
   \\
   (\exRec{f}{\sigma}) \join (\exRec{f}{\sigma'}) &= \exRec{f}{(\sigma \join \sigma')}
   \\
   (\exApp{e_1}{e_2}) \join (\exApp{e_1'}{e_2'}) &= \exApp{(e_1 \join e_1')}{(e_2 \join e_2')}
   \\
   \exNil \join \exNil &= \exNil
   \\
   \exCons{e_1}{e_2} \join \exCons{e_1'}{e_2'} &= \exCons{e_1 \join e_1'}{e_2 \join e_2'}
   \\
   \exConsProjL{e} \join \exConsProjL{e'} &= \exConsProjL{e \join e'}
   \\
   \exConsProjR{e} \join \exConsProjR{e'} &= \exConsProjR{e \join e'}
   \\
   \exConsProjL{e} \join \exConsProjR{e'} &= \exCons{e}{e'}
   \\
   \exConsProjR{e} \join \exConsProjL{e'} &= \exCons{e'}{e}
\end{salign}
\caption{Join of term slices}
\end{figure}
