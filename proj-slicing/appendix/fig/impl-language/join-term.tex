\shadebox{$e \join e'$}
\begin{salign}
   \bot \join e &= e
   \\
   e \join \bot &= e
   \\
   \exVar{x} \join \exVar{x} &= \exVar{x}
   \\
   \exTrue \join \exTrue &= \exTrue
   \\
   \exTrue \join \exTrueSel &= \exTrueSel
   \\
   \exFalse \join \exFalse &= \exFalse
   \\
   \exFalse \join \exFalseSel &= \exFalseSel
   \\
   \exInt{n} \join \exInt{n} &= \exInt{n}
   \\
   \exInt{n} \join \exIntSel{n} &= \exIntSel{n}
   \\
   \exPair{e_1}{e_2} \join \exPair{e_1'}{e_2'} &= \exPair{e_1 \join e_1'}{e_2 \join e_2'}
   \\
   \exPair{e_1}{e_2} \join \exPairSel{e_1'}{e_2'} &= \exPairSel{e_1 \join e_1'}{e_2 \join e_2'}
   \\
   (\exRec{f}{\sigma}) \join (\exRec{f}{\sigma'}) &= \exRec{f}{(\sigma \join \sigma')}
   \\
   (\exApp{e_1}{e_2}) \join (\exApp{e_1'}{e_2'}) &= \exApp{(e_1 \join e_1')}{(e_2 \join e_2')}
   \\
   \exNil \join \exNil &= \exNil
   \\
   \exNil \join \exNilSel &= \exNilSel
   \\
   \exCons{e_1}{e_2} \join \exCons{e_1'}{e_2'} &= \exCons{e_1 \join e_1'}{e_2 \join e_2'}
   \\
   \exCons{e_1}{e_2} \join \exConsSel{e_1'}{e_2'} &= \exConsSel{e_1 \join e_1'}{e_2 \join e_2'}
\end{salign}
