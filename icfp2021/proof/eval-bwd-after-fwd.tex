   \matchBwd{\rho_3'}{e^\twoPrime}{\beta'}{w}{u}{\sigma'}
   \leq
   v, \sigma
   &

   \locallabel{apply-premise-two}
   \\
   &
   \evalBwdLeq{\exClosure{\rho_1}{h}{\sigma}}{T}{\rho}{e_1}{\alpha}
