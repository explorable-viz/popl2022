\begin{proof}

\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &

         lab={\ruleName{$\evalBwdS$-var}}

      {
         \evalBwd{v}{\trVar{x}{\Gamma}}{\rho}{\exVar{x}}{\FF}
      }
   \end{smathpar}
   }
   &
   \notag
   \\
   &
   \envLookup{\rho}{x}{v}

   \text{
   	(\lemref{core-language:env-get-put})
   }
   \notag

   \qedLocal

         lab={\ruleName{$\evalFwdS$-var}}
      ]

      }
      {
         \evalFwd{\rho}{x}{\FF}{\trVar{x}{\Gamma}}{v}

   }
   &
   \notag
   \\
   &
   \qedLocal
   \TT
