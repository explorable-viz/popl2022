}

}

\and

{
   \Gamma \vdash \kappa: B
   \\
   \Gamma \vdash \sigma: \tyFun{A}{\tyFun{\tyList{A}}{B}}

{
   \Gamma \vdash \elimList{\kappa}{\sigma}: \tyFun{\tyList{A}}{B}
}
