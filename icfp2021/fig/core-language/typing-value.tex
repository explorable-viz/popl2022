\flushleft \shadebox{$\vdash \rho: \Gamma$}
\begin{smathpar}
\inferrule*
{
   \strut
}
{
   \vdash \envEmpty: \cxtEmpty
}
%
\and
%
\inferrule*
{
   \vdash \rho: \Gamma
   \\
   \vdash v: A
}
{
   \vdash \envExtend{\rho}{k}{v}: \cxtExtend{\Gamma}{k}{A}
}
\end{smathpar}
\\[2mm]
\flushleft \shadebox{$\Gamma \vdash h: \Delta$}
\begin{smathpar}
   \inferrule*[right={
      \textnormal{$\Delta = \seqRange{x_1: \tyFun{A_1}{B_1}}{x_n: \tyFun{A_n}{B_n}}$}
   }]
   {
      \Gamma \concat \Delta \vdash \sigma_i: \tyFun{A_i}{B_i}
      \quad
      (\forall i \in \setRange{1}{h})
   }
   {
      \Gamma \vdash \seqRange{x_1 : \sigma_1}{x_n : \sigma_n}: \Delta
   }
\end{smathpar}
\\[2mm]
\flushleft \shadebox{$\vdash v: A$}
\begin{smathpar}
\inferrule*
{
   \strut
}
{
   \vdash \exInt{n}: \tyInt
}
%
\and
%
\inferrule*[
   right={$\phi \in \Int \to \Int$}
]
{
   \strut
}
{
   \vdash \exPrim{\phi}: \tyFun{\tyInt}{\tyInt}
}
%
\and
%
\inferrule*[
   right={$\phi \in \Int \to \Int \to \Int$}
]
{
   \strut
}
{
   \vdash \exPrim{\phi}: \tyFun{\tyInt}{\tyFun{\tyInt}{\tyInt}}
}
%
\and
%
\inferrule*
{
   \strut
}
{
   \vdash \exTrue: \tyBool
}
%
\and
%
\inferrule*
{
   \strut
}
{
   \vdash \exFalse: \tyBool
}
%
\and
%
\inferrule*
{
   \vdash v: A
   \\
   \vdash v': B
}
{
   \vdash \exPair{v}{v'}: \tyProd{A}{B}
}
%
\and
%
\inferrule*
{
   \vdash \rho: \Gamma
   \\
   \Gamma \vdash h: \Delta
   \\
   \Gamma \concat \Delta \vdash \sigma: \tyFun{A}{B}
}
{
   \vdash \exClosureRec{\Gamma}{\rho}{h}{\sigma}: \tyFun{A}{B}
}
%
\and
%
\inferrule*
{
   \strut
}
{
   \vdash \exNil: \tyList{A}
}
%
\and
%
\inferrule*
{
   \vdash v: A
   \\
   \vdash v': \tyList{A}
}
{
   \vdash (\exCons{v}{v'}): \tyList{A}
}
%
\and
%
\inferrule*
{
   \forall (i,j) \leq (i',j').
   \vdash v_{i,j}: A
}
{
   \vdash \exMatrix{v_{i,j}}{i}{j}{\exPair{i'}{j'}}: \tyMatrix{A}
}
