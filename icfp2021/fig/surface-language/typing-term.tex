\flushleft \shadebox{$\Gamma \vdash s: A$}
\begin{smathpar}
\inferrule*
{
   \Gamma \vdash s : A
   \\
   \Gamma \vdash \overrightarrow{s} : \tyList{A}
}
{
   \Gamma \vdash s \concat \overrightarrow{s} : \tyList{A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash s: A
   \\
   \Gamma \vdash \sigma: \tyFun{A}{B}
}
{
   \Gamma \vdash \exMatch{s}{\sigma}: B
}
%
\and
%
\inferrule*
{
   \Gamma \vdash s: \tyBool
   \\
   \Gamma \vdash s': A
   \\
   \Gamma \vdash s^{\twoPrime}: A
}
{
   \Gamma \vdash \exIfThenElse{s}{s'}{s^{\twoPrime}}: A
}
%
\and
%
\inferrule*
{
   \strut
}
{
   \Gamma \vdash \sExNil : \kw{List A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash s : \kw{A}
   \\
   \Gamma \vdash \overrightarrow{s} : \kw{List A}
}
{
   \Gamma \vdash \sExCons{s}{\overrightarrow{s}} : \kw{List A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash s: \tyInt
   \\
   \Gamma \vdash s': \tyInt
}
{
   \Gamma \vdash \exListSeq{s}{s'}: \tyList{\tyInt}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash \overrightarrow{q} \dashv \Delta
   \\
   \Gamma \concat \Delta \vdash s: A
}
{
   \Gamma \vdash \exListComp{s}{\overrightarrow{q}}: \tyList{A}
}
\end{smathpar}

\end{figure}
