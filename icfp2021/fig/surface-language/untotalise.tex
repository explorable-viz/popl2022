\flushleft\shadebox{untotalise p $\sigma = \kappa$}
\begin{salign}
   \untotalise{x}{\elimVar{x}{\kappa}} &= \kappa
   \\
   \untotalise{\pattTrue}{\elimBool{\kappa}{\kappa'}} &= \kappa
   \\
   \untotalise{\pattFalse}{\elimBool{\kappa}{\kappa'}} &= \kappa'
   \\
   \untotalise{\pattPair{p_1}{p_2}}{\elimProd{\sigma}} &= \untotalise{p_2}{\untotalise{p_1}{\sigma}}
   \\
   \untotalise{\pattNil}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma}}} &= \kappa
   \\
   \untotalise{\pattCons{p}{p'}}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma}}} &= \untotalise{p'}{\untotalise{p}{\sigma}}
   \\
   \untotalise{\pattList{p}{o}}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma}}} &= \untotaliseRest{o}{\untotalise{p}{\sigma}}
   \\
\end{salign}

\flushleft\shadebox{untotaliseRest o $\sigma = \kappa$}
\begin{salign}
   \untotaliseRest{\sExNil}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma}}}
      &= \kappa
   \\
   \untotaliseRest{\sExCons{p}{o}}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma}}}
      &= \untotaliseRest{o}{\untotalise{p}{\sigma}}
\end{salign}
