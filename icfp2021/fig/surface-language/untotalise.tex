\flushleft\shadebox{untotalise $\sigma \ \sigma' = \sigma''$}
\begin{salign}
   \untotalise{\elimVar{x}{\kappa}}{\elimVar{x}{\kappa'}} &= \elimVar{x}{\kappa'}
   \\
   \untotalise{\elimBoolTrue{\kappa}}{\elimBool{\kappa_1}{\kappa_2}} &= \elimBoolTrue{\kappa_1}
   \\
   \untotalise{\elimBoolFalse{\kappa}}{\elimBool{\kappa_1}{\kappa_2}} &= \elimBoolFalse{\kappa_2}
   \\
   \untotalise{\elimProd{\sigma}}{\elimProd{\sigma'}} &= \elimProd{\sigma'}
   \\
   \untotalise{\branchNil{\kappa}}{\elimList{\branchNil{\kappa'}}{\branchCons{\sigma}}} &= \branchNil{\kappa'}
   \\
   \untotalise{\branchCons{\sigma}}{\elimList{\branchNil{\kappa}}{\branchCons{\sigma'}}} &= \branchCons{\sigma'}
\end{salign}
\caption{Untotalise - Match partial eliminator against total eliminator}
