\begin{theorem}[Galois connection for desugaring]
  \label{thm:surface-language:desugar:gc}
     Suppose $s \desugarFwd e$. Then:
  \begin{enumerate}
     \item $\desugarFwdF{s}(\desugarBwdF{s}(e')) \geq e'$.
     \item $\desugarBwdF{s}(\desugarFwdF{s}(s')) \leq s'$.
  \end{enumerate}
\end{theorem}
