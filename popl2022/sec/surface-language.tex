\begin{definition}[Forward and backward functions for desugaring]
     Suppose $s \desugarFwd e$. Define $\desugarFwdF{s}: \Below{s} \to \Below{e}$ and $\desugarBwdF{s}: \Below{e} \to \Below{s}$ to be $\desugarFwd$ and $\desugarBwd{s}$ domain-restricted to $\Below{s}$ and $\Below{e}$ respectively.
\end{definition}

\begin{theorem}[Galois connection for desugaring]
  \label{thm:surface-language:desugar:gc}
     Suppose $s \desugarFwd e$. Then $\desugarFwdF{s} \adjoint \desugarBwdF{s}$.
\end{theorem}
