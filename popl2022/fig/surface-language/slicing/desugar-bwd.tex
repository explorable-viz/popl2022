$\begin{array}{lll|llll}
   \arrayrulecolor{lightgray}
   &&&\textit{where}
   \\
   \rowcolor{verylightgray}
   \desugarBwdF{s} &::& \Below{s} \to \Below{e}
   & s \desugarFwd e
   &&
   \\
   % hole
   \desugarBwdF{t}(e)
   &=&
   \desugarBwdF{t}(e')
   &
   e \eq e'
   \\
   % op
   \desugarBwdF{\exOp{\primOp}}(\primOp)
   &=&
   \exOp{\primOp}
   \\
   % binary-apply
   \desugarBwdF{\exBinaryApp{t}{\primOp}{t'}}(\exApp{\exApp{\primOp}{e}}{e'})
   &=&
   \exBinaryApp{\desugarBwdF{t}(e)}{\primOp}{\desugarBwdF{t'}(e')}
   \\
   % nil
   \desugarBwdF{\exNil}(\annNil{\alpha}) &=& \annNil{\alpha}
   \\
   % non-empty-list
   \desugarBwdF{\exList{t}{l}}(\annCons{e}{e'}{\alpha})
   &=&
   \annList{\desugarBwdF{t}(e)}{\desugarBwdF{l}(e')}{\alpha}
   \\
   % cons
   \desugarBwdF{\exCons{t}{t'}}(\annCons{e}{e'}{\alpha})
   &=&
   \annCons{\desugarBwdF{t}(e)}{\desugarBwdF{t'}(e')}{\alpha}
   \\
   % let-rec
   \desugarBwdF{\exLetRecMutual{\vec{g}}{t}}
               (\exLetRecMutual{\vec{\bind{x}{\sigma}}}{e'})
   &=&
   \exLetRecMutual{\seqRange{\bind{x_1}{\vec{c_1}'}}{\bind{x_j}{\vec{c_j}'}}}{s}
   &
   g_i &=& \bind{x_i}{\vec{c_i}}
   \\
   &&&
   \vec{c_i}' &=& \desugarClausesBwdF{\vec{c_i}}(\sigma_i)
   \\
   &&&
   s &=& \desugarBwdF{t}(e')
   \\
   % apply
   \desugarBwdF{\exApp{t}{t'}}(\exApp{e}{e'})
   &=&
   \exApp{\desugarBwdF{t}(e)}{\desugarBwdF{t'}(e')}
   \\
   % if
   \desugarBwdF{\exIfThenElse{t}{t_1}{t_2}}(\exApp{\exLambda{\sigma}}{e})
   &=&
   \exIfThenElse{\desugarBwdF{t}(e)}{s_1}{s_2}
   &
   \sigma &=& \elimBool{e_1}{e_2}
   \\
   &&&
   s_1 &=& \desugarBwdF{t_1}(e_1)
   \\
   &&&
   s_2 &=& \desugarBwdF{t_2}(e_2)
   \\
   % match
   \desugarBwdF{\exMatch{t}{\vec{c}}}(\exApp{\exLambda{\sigma}}{e})
   &=&
   \exMatch{\desugarBwdF{t}(e)}{\desugarClausesBwdF{\vec{c}}(\sigma)}
   \\
   % let
   \desugarBwdF{\exLet{p}{t}{t'}}(\exApp{\exLambda{\sigma}}{e})
   &=&
   \exLet{p}{\desugarBwdF{t}(e)}{s'}
   &
   s' &=& \desugarClausesBwdF{\clauseUncurried{p}{t'}}(\sigma)
   \\
   % list-enum
   \desugarBwdF{\exListEnum{t}{t'}}(\exApp{\exApp{\funEnumFromTo}{e}}{e'})
   &=&
   \exListEnum{\desugarBwdF{t}(e)}{\desugarBwdF{t'}(e')}
   \\
   % list-comp-done
   \desugarBwdF{\exListComp{t}{\seqEmpty}}(\annCons{e}{\annNil{\alpha}}{\alpha'})
   &=&
   \annot{\exListComp{\desugarBwdF{t}(e)}{\seqEmpty}}{\alpha \join \alpha'}
   \\
   % list-comp-guard
   \desugarBwdF{\exListComp{t}{\qualGuard{t'} \concat \vec{q}}}
              (\exApp{\exLambda{\sigma}}{e})
   &=&
   \annot{\exListComp{s'}{\qualGuard{\desugarBwdF{t'}(e)} \concat \vec{q}'}}{\alpha \join \beta}
   &
   \sigma &=& \elimBool{e'}{\annNil{\alpha}}
   \\
   &&&
   \annListComp{s'}{\vec{q}'}{\beta} &=& \desugarBwdF{\exListComp{t'}{\vec{q}}}(e')
   \\
   % list-comp-decl
   \desugarBwdF{\exListComp{t}{\qualDeclaration{p}{t'} \concat \vec{q}}}
               (\exApp{\exLambda{\sigma}}{e})
   &=&
   \annot{\exListComp{s}{\qualDeclaration{p}{\desugarBwdF{t'}(e)} \concat \vec{q}'}}
         {\beta}
   &
   \clauseUncurried{p}{\annot{\exListComp{s}{\vec{q}'}}{\beta}}
   &=&
   \desugarClausesBwdF{\clauseUncurried{p}{\exListComp{t}{\vec{q}}}}(\sigma)
   \\
   % list-comp-gen
   \desugarBwdF{\exListComp{s_{2}}{\qualGenerator{p}{s_{1}} \concat \vec{q}}}
               (\exApp{e}{e_1})
   &=&
   \annot{\exListComp{s_{2}'}{\qualGenerator{p}{\desugarBwdF{s_1}(e_{1})} \concat \vec{q}'}}
         {\beta \join \beta'}
   &
   e &=& \exApp{\funConcatMap}{\exLambda{\sigma}}
   \\
   &&&
   (\sigma',\beta) &=& \totaliseBwdF{p}(\sigma)
   \\
   &&&
   \clauseUncurried{p}{\annot{\exListComp{s_{2}'}{\vec{q}'}}{{\beta'}}}
   &=&
   \desugarClausesBwdF{\clauseUncurried{p}{\exListComp{s_2}{\vec{q}}}}(\sigma')
   \\
\end{array}$
\caption{Desugaring terms --- backward slicing}
\end{figure}
