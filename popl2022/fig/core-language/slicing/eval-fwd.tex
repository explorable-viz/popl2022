      \strut
   }

      \evalFwd{\rho}

              {\trLambda{\sigma'}}
              {\annClosure{\rho}{\seqEmpty}{\alpha}{\sigma}}

   \and
   %
   \inferrule*[lab={\ruleName{$\evalFwdS$-int}}]

      \strut
