\inferrule*
{
   \Gamma \vdash e: \tyFun{A}{B}
   \\
   \Gamma \vdash e': A

{
   \Gamma \vdash \exApp{e}{e'}: B
}

}

{
   \Gamma \vdash e: A
   \\
   \Gamma \vdash e': \tyList{A}
}
{
   \Gamma \vdash (\exCons{e}{e'}): \tyList{A}
}
%
\and
%
\inferrule*
{
   \Gamma \vdash e': \tyInt
   \\
   \Gamma \concat \bind{x}{\tyInt} \vdash e: A

{
   \Gamma \vdash \exVec{e}{x}{e'}: \tyVec{A}
}

\and

{
   \Gamma \vdash e: \tyVec{A}

   \Gamma \vdash e': \tyInt
}
{
   \Gamma \vdash \exVecLookup{e}{e'}: A
}

\and

{
   \Gamma \vdash e: \tyVec{A}
}
{
   \Gamma \vdash \exVecLen{e}: \tyInt
}
%
\and

{
   \Gamma \vdash h: \Delta
   \\
   \Gamma \concat \Delta \vdash e: A
}
{
   \Gamma \vdash \exLetRecMutual{h}{e}: A
