\subsection{\thmref{surface-language:desugar:gc}, part (i)}
Induction on the $\dbwdarrow$ derivation.
\begin{proof}
   Suppose $s \dfwdarrow e$. Then $\desugarFwdF{s}(\desugarBwdF{s}(e')) \geq e'$.
\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-binary-apply}}
         ]
         {
            e \eq \exApp{\exApp{\exOp{\primOp}}{e_{1}}}{e_{2}}
            \\
            e_1, s_1 \dbwdarrow s_1'
            \\
            e_2, s_2 \dbwdarrow s_2'
         }
         {
            e, \exBinaryApp{s_{1}}{\primOp}{s_{2}}
            \dbwdarrow
            \exBinaryApp{s_{1}'}{\primOp}{s_{2}'}
         }
      \end{smathpar}
   }
   &
   \notag
   \\
   &
   s_{1}' \dfwdarrow e_{1}' \geq e_{1} \quad (\exists e_{1}')
   &
   \text{(IH)}
   \locallabel{dfb:binary-ih1}
   \\
   &
   s_{2}' \dfwdarrow e_{2}' \geq e_{2} \quad (\exists e_{2}')
   &
   \text{(IH)}
   \locallabel{dfb:binary-ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
   \begin{smathpar}
      \inferrule*[
         lab={\ruleName{$\dfwdarrow$-binary-apply}}
      ]
      {
         s_{1}' \dfwdarrow e_{1}'
         \\
         s_{2}' \dfwdarrow e_{2}'
      }
      {
         \exBinaryApp{s_{1}'}{\primOp}{s_{2}'} \dfwdarrow \exApp{\exApp{\exOp{\primOp}}{e_{1}'}}{e_{2}'}
      }
   \end{smathpar}
   }
   &
   \text{(\localref{dfb:binary-ih1}, \localref{dfb:binary-ih2})}
   \notag
   \\
   &
   \qedLocal
   \exApp{\exApp{\exOp{\primOp}}{e_{1}'}}{e_{2}'} \geq \exApp{\exApp{\exOp{\primOp}}{e_{1}}}{e_{2}}
   &
   \text{(\localref{dfb:binary-ih1}, \localref{dfb:binary-ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-nil}}
         ]
         {
            e \eq \annNil{\alpha}
         }
         {
            e, \exNil \dbwdarrow \annNil{\alpha}
         }
      \end{smathpar}
   }
   &
   \locallabel{dfb:nil}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
   \begin{smathpar}
      \inferrule*[
         lab={\ruleName{$\dfwdarrow$-nil}}
      ]
      {
         \strut
      }
      {
         \annNil{\alpha} \dfwdarrow \annNil{\alpha}
      }
   \end{smathpar}
   }
   &
   \notag
   \\
   &
   \qedLocal
   e \eq \annNil{\alpha} \Rightarrow \annNil{\alpha} \geq e
   &
   \text{(\localref{dfb:nil})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-non-empty-list}}
         ]
         {
            e \eq \annCons{e_1}{e_2}{\alpha}
            \\
            e_1, s \dbwdarrow s'
            \\
            e_2, l \dbwdarrow l'
         }
         {
            e, \exList{s}{l}
            \dbwdarrow
            \annList{s'}{l'}{\alpha}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   s' \dfwdarrow e_{1}' \geq e_{1} \quad (\exists e_{1}')
   &
   \text{(IH)}
   \locallabel{dfb:nelist-ih1}
   \\
   &
   l' \dfwdarrow e_{2}' \geq e_{2} \quad (\exists e_{2}')
   &
   \text{(IH)}
   \locallabel{dfb:nelist-ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dfwdarrow$-non-empty-list}}
         ]
         {
            s' \dfwdarrow e_{1}'
            \\
            l' \dfwdarrow e_{2}'
         }
         {
            \annList{s'}{l'}{\alpha} \dfwdarrow \annCons{e_{1}'}{e_{2}'}{\alpha}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{dfb:nelist-ih1}, \localref{dfb:nelist-ih2})}
   \notag
   \\
   &
   \qedLocal
   \annCons{e_{1}'}{e_{2}'}{\alpha} \geq \annCons{e_1}{e_2}{\alpha}
   &
   \text{(\localref{dfb:nelist-ih1}, \localref{dfb:nelist-ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-cons}}
         ]
         {
            e \eq \annCons{e_1}{e_2}{\alpha}
            \\
            e_1, s_1 \dbwdarrow s_1'
            \\
            e_2, s_2 \dbwdarrow s_2'
         }
         {
            e, \exCons{s_1}{s_2}
            \dbwdarrow
            \annCons{s_1'}{s_2'}{\alpha}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   s_{1}' \dfwdarrow e_{1}' \geq e_{1} \quad (\exists e_{1}')
   &
   \text{(IH)}
   \locallabel{dfb:cons-ih1}
   \\
   &
   s_{1}' \dfwdarrow e_{1}' \geq e_{1} \quad (\exists e_{1}')
   &
   \text{(IH)}
   \locallabel{dfb:cons-ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dfwdarrow$-cons}}
         ]
         {
            s_{1}' \dfwdarrow e_{1}'
            \\
            s_{2}' \dfwdarrow e_{2}'
         }
         {
            \annCons{s_{1}'}{s_{2}'}{\alpha} \dfwdarrow \annCons{e_{1}'}{e_{2}'}{\alpha}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{dfb:cons-ih1}, \localref{dfb:cons-ih2})}
   \notag
   \\
   &
   \qedLocal
   \annCons{e_{1}'}{e_{2}'}{\alpha} \geq \annCons{e_1}{e_2}{\alpha}
   &
   \text{(\localref{dfb:cons-ih1}, \localref{dfb:cons-ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-let-rec}}
         ]
         {
            e \eq \exLetRecMutual{\seqRange{\bind{x_1}{\sigma_1}}{\bind{x_j}{\sigma_j}}}{e_1}
            \\
            \sigma_i, \vec{c_i} \dbwdarrow \vec{c_i}'
            \quad
            (\forall i \numleq j)
            \\
            e_1, s \dbwdarrow s'
         }
         {
            e, \exLetRecMutual{\seqRange{\bind{x_1}{\vec{c}_1}}{\bind{x_j}{\vec{c_j}}}}{s}
            \dbwdarrow
            \exLetRecMutual{\seqRange{\bind{x_1}{\vec{c_1}'}}{\bind{x_j}{\vec{c_j}'}}}{s'}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   \vec{c_i}' \dfwdarrow \sigma_{i}' (\forall i \numleq j) \geq \sigma_{i} (\forall i \numleq j) \quad (\exists \sigma_{i}')
   \locallabel{dfb:letrec:ih1}
   \\
   &
   s' \dfwdarrow e' \geq e \quad (\exists e')
   \locallabel{dfb:letrec:ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dfwdarrow$-let-rec}}
         ]
         {
            \vec{c_i}' \dfwdarrow \sigma_{i}'
            \quad
            (\forall i \numleq j)
            \\
            s' \dfwdarrow e'
         }
         {
            \exLetRecMutual{\seqRange{x_1: \vec{c_1}'}{x_j: \vec{c_j}'}}{s'}
            \dfwdarrow
            \exLetRecMutual{\seqRange{x_1: \sigma_1'}{x_j: \sigma_j'}}{e'}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{dfb:letrec:ih1}, \localref{dfb:letrec:ih2})}
   \notag
   \\
   &
   \qedLocal
   \exLetRecMutual{\seqRange{x_1: \sigma_1'}{x_j: \sigma_j'}}{e'} \geq \exLetRecMutual{\seqRange{\bind{x_1}{\sigma_1}}{\bind{x_j}{\sigma_j}}}{e_1}
   &
   \text{(\localref{dfb:letrec:ih1}, \localref{dfb:letrec:ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-apply}}
         ]
         {
            e \eq \exApp{e_1}{e_2}
            \\
            e_1, s_1 \dbwdarrow s_1'
            \\
            e_2, s_2 \dbwdarrow s_2'
         }
         {
            e, \exApp{s_1}{s_2}
            \dbwdarrow
            \exApp{s_1'}{s_2'}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   s_{1}' \dfwdarrow e_{1}' \geq e_{1} \quad (\exists e_{1}')
   \locallabel{dfb:app:ih1}
   \\
   &
   s_{2}' \dfwdarrow e_{2}' \geq e_{1} \quad (\exists e_{2}')
   \locallabel{dfb:app:ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dfwdarrow$-apply}}
         ]
         {
            s_{1}' \dfwdarrow e_{1}'
            \\
            s_{2}' \dfwdarrow e_{2}'
         }
         {
            \exApp{s_{1}'}{s_{2}'} \dfwdarrow \exApp{e_{1}'}{e_{2}'}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{dfb:app:ih1}, \localref{dfb:app:ih2})}
   \notag
   \\
   &
   \qedLocal
   \exApp{e_{1}'}{e_{2}'} \geq \exApp{e_{1}}{e_{2}}
   &
   \text{(\localref{dfb:app:ih1}, \localref{dfb:app:ih2})}
   \notag
\end{flalign}
\end{proof}
