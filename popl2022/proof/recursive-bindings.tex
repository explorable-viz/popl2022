\subsection{\thmref{core-language:closeDefs:gc}, part (i)}
\flushleft\shadebox{$\rho \closeDefsBwdR \rho', h$}
\begin{salign}
   &\vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
\end{salign}
\flushleft\shadebox{$\rho, h \closeDefsR \rho'$}
\begin{salign}
   &\rho, h
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   & \text{where $h = \vec{\bind{x}{\sigma}}$}
\end{salign}
\vspace{5pt}
Suppose $\rho, h \closeDefsR \rho'$.  Then $\closeDefsFwdF{\rho,h}(\closeDefsBwdF{\rho,h}(\rho')) \geq \rho'$.
\begin{proof}
\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   &
   \text{(Suppose)}
   \notag
   \\
   &
   \vec{h} = \seqRange{\vec{\bind{x_1}{\sigma_1'}}}{\vec{\bind{x_j}{\sigma_j'}}}
   &
   \notag 
   \\
   &
   \sigma''_i = \sigma_{1, i}' \join .. \join \sigma_{j, i}'
   &
   \notag
   \\
   &
   \tau_i = \sigma_i \join \sigma_i''
   &
   \notag
   \\
   &
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\tau_1}}}
            {\bind{x_j}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\tau_j}}}
   &
   \notag
   \\
   &
   \bigjoin\vec{\rho} \geq \rho
   &
   \locallabel{recbindbtf:1}
   \\
   &
   \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}} \geq h
   &
   \locallabel{recbindbtf:2}
   \\
   &
   \sigma_i \join \sigma_i'' \geq \sigma_i
   &
   \locallabel{recbindbtf:3}
   \\
   &
   \qedLocal
   \seqRange{\bind{x_1}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\tau_1}}}
   {\bind{x_j}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\tau_j}}}
   \geq
   \vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   &
   \text{(\localref{recbindbtf:1}, \localref{recbindbtf:2}, \localref{recbindbtf:3})}
   \notag
\end{flalign}
\end{proof}


\subsection{\thmref{core-language:closeDefs:gc}, part (ii)}
Suppose $\rho, h \closeDefsR \rho'$.  Then $\closeDefsBwdF{\rho,h}(\closeDefsFwdF{\rho,h}(\rho, h)) \leq (\rho, h)$.
\begin{proof}
\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \rho, h
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   \hspace{5pt}
   \text{where $h = \vec{\bind{x}{\sigma}}$}
   &
   \notag
   \\
   &
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   &
   \\
   &
   \qedLocal
   \rho, h \leq (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   \notag
\end{flalign}
\end{proof}
