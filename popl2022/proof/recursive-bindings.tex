\subsection{\thmref{core-language:closeDefs:gc}, part (i)}
\flushleft\shadebox{$\rho \closeDefsBwdR \rho', h$}
\begin{salign}
   &\vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
\end{salign}
\flushleft\shadebox{$\rho, h \closeDefsR \rho'$}
\begin{salign}
   &\rho, h
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   & \text{where $h = \vec{\bind{x}{\sigma}}$}
\end{salign}
\vspace{5pt}
\begin{proof}
   Suppose $\rho, h \closeDefsR \rho'$.  Then $\closeDefsFwdF{\rho,h}(\closeDefsBwdF{\rho,h}(\rho')) \geq \rho'$.
\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   &
   \notag
   \\
   &
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\sigma_1}}}
            {\bind{x_j}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\sigma_j}}}
   &
   \notag
   \\
   &
   \qedLocal
   \seqRange{\bind{x_1}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\sigma_1}}}
   {\bind{x_j}{\exClosure{\bigjoin\vec{\rho}}{\vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}}}{\sigma_j}}}
   \geq
   \vec{\bind{x}{\exClosure{\rho}{h}{\sigma}}}
   &
   \notag
\end{flalign}
\end{proof}


\subsection{\thmref{core-language:closeDefs:gc}, part (ii)}
\begin{proof}
   Suppose $\rho, h \closeDefsR \rho'$.  Then $\closeDefsBwdF{\rho,h}(\closeDefsFwdF{\rho,h}(\rho, h)) \leq (\rho, h)$.
\small
\begin{flalign}
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \rho, h
   \closeDefsR
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   \hspace{5pt}
   \text{where $h = \vec{\bind{x}{\sigma}}$}
   &
   \notag
   \\
   &
   \seqRange{\bind{x_1}{\exClosure{\rho}{h}{\sigma_1}}}{\bind{x_j}{\exClosure{\rho}{h}{\sigma_j}}}
   \closeDefsBwdR
   (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   &
   \\
   &
   \qedLocal
   \rho, h \leq (\bigjoin\vec{\rho}, \vec{\bind{x}{\sigma}} \join {\bigjoin\vec{h}})
   \notag
\end{flalign}
\end{proof}
