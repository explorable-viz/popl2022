   \annListComp{s'}{\qualDeclaration{p}{s_1'}}{\beta \join \beta'} \leq \annListComp{s}{\qualDeclaration{p}{s_{1}}}{\alpha}
   &
   \text{(\localref{listcomplast:ih1}, \localref{listcomplast:ih2}, \localref{listcomplast:ih3})}
   \notag
   \\
   &
   \notag
   \intertext{The cases when the final qualifier is a guard or a generator are similar.}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\desugarFwd$-list-comp-guard}}
         ]
         {
            \annListComp{s_{1}}{\vec{q}}{\alpha}
            \desugarFwd e_{1}
            \\
            s_{2} \desugarFwd e_{2}
         }
         {
            \annListComp{s_{1}}{\qualGuard{s_{2}} \concat \vec{q}}{\alpha}
            \desugarFwd
            \exApp{\smash{\exLambda{\elimBool{e_{1}}{\annNil{\alpha}}}}}{e_{2}}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   e_2, s_2 \dbwdarrow s_2' \leq s_{2} \quad (\exists s_{2}')
   &
   \text{(IH)}
   \locallabel{listguard:ih1}
   \\
   &
   e_1, \exListComp{s_1}{\vec{q}} \dbwdarrow \annListComp{s_{1}'}{\vec{q}'}{\beta} \leq \annot{\exListComp{s_1}{\vec{q}}}{\alpha} \quad (\exists s_{1}', \vec{q}', \beta)
   &
   \text{(IH)}
   \locallabel{listguard:ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-list-comp-guard}},
         ]
         {
            e \eq \exApp{\exLambda{\elimBool{e_{1}}{\annNil{\alpha}}}}{e_{2}}
            \\
            e_2, s_2 \dbwdarrow s_2'
            \\
            e_1, \exListComp{s_1}{\vec{q}} \dbwdarrow \annListComp{s_{1}'}{\vec{q}'}{\beta}
         }
         {
            e,
            \exListComp{s_1}{\qualGuard{s_2} \concat \vec{q}}
            \dbwdarrow
            \annot{\exListComp{s_{1}'}{\qualGuard{s_{2}'} \concat \vec{q}'}}{\alpha \join \beta}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{listguard:ih1}, \localref{listguard:ih2})}
   \notag
   \\
   &
   \qedLocal
   \annot{\exListComp{s_{1}'}{\qualGuard{s_{2}'} \concat \vec{q}'}}{\alpha \join \beta} \leq \annListComp{s_{1}}{\qualGuard{s_{2}} \concat \vec{q}}{\alpha}
   &
   \text{(\localref{listguard:ih1}, \localref{listguard:ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\desugarFwd$-list-comp-decl}}
         ]
         {
            \clauseUncurried{p}{\annListComp{s_{2}}{\vec{q}}{\alpha}} \desugarFwd \sigma
            \\
            s_{1} \desugarFwd e_{1}
         }
         {
            \annListComp{s_{2}}{\qualDeclaration{p}{s_{1}} \concat \vec{q}}{\alpha}
            \desugarFwd
            \exApp{(\exLambda{\sigma})}{e_{1}}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   \sigma, \clauseUncurried{p}{\exListComp{s_{2}}{\vec{q}}}
            \dbwdarrow
            \clauseUncurried{p}{\annot{\exListComp{s_{2}'}{\vec{q}'}}{\beta}}\leq \clauseUncurried{p}{\annot{\exListComp{s_{2}}{\vec{q}}}{\alpha} } \quad (\exists s_{2}', \vec{q}', \beta)
   &
   \text{(\ref{lem:aux:clause:gc})}
   \locallabel{listdecl:ih1}
   \\
   &
   e_{1}, s_{1} \dbwdarrow s_{1}' \leq s_{1} \quad (\exists s_{1}')
   &
   \text{(IH)}
   \locallabel{listdecl:ih2}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-list-comp-decl}},
         ]
         {
            e \eq \exApp{(\exLambda{\sigma})}{e_1}
            \\
            \sigma, \clauseUncurried{p}{\annot{\exListComp{s_{2}}{\vec{q}}}{\alpha}}
            \dbwdarrow
            \clauseUncurried{p}{\annot{\exListComp{s_{2}'}{\vec{q}'}}{\beta}}
            \\
            e_{1}, s_{1} \dbwdarrow s_{1}'
         }
         {
            e,
            \exListComp{s_2}{\qualDeclaration{p}{s_1} \concat \vec{q}}
            \dbwdarrow
            \annot{\exListComp{s_{2}'}{\qualDeclaration{p}{s_{1}'} \concat \vec{q}'}}{\beta}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{listdecl:ih1}, \localref{listdecl:ih2})}
   \notag
   \\
   &
   \qedLocal
   \annot{\exListComp{s_{2}'}{\qualDeclaration{p}{s_{1}'} \concat \vec{q}'}}{\beta} \leq \annListComp{s_{2}}{\qualDeclaration{p}{s_{1}} \concat \vec{q}}{\alpha}
   &
   \text{(\localref{listdecl:ih1}, \localref{listdecl:ih2})}
   \notag
   \\
   \intertext{\crossrule}
   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   &
   \caseDerivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\desugarFwd$-list-comp-gen}}
         ]
         {
            \inferrule*[]
            {
               \annot{\exListComp{s_{2}}{\vec{q}}}{\alpha} \desugarFwd e_2
               \\
               p, e_2 \desugarFwd \sigma_1
            }
            {
               \annot{\clauseUncurried{p}{\exListComp{s_{2}}{\vec{q}}}}{\alpha} \desugarFwd \sigma_{1}
            }
            \\
            \totalise{\sigma_{1}}{\alpha}{\sigma_{2}}
            \\
            s_{1} \desugarFwd e_{1}
         }
         {
            \annListComp{s_{2}}{\qualGenerator{p}{s_{1}} \concat \vec{q}}{\alpha}
            \desugarFwd
            \exApp{\exApp{\funConcatMap}{(\exLambda{\sigma_{2}})}}{e_{1}}
         }
      \end{smathpar}
   }
   \notag
   \\
   &
   e_1, s_1 \dbwdarrow s_1' \quad (\exists s_1')
   &
   \text{(IH)}
   \locallabel{listgen:ih1}
   \\
   &
   \totalisebwd{\sigma_2}{p}{\sigma_1'}{\beta} \leq \sigma_1, \alpha \quad (\exists \sigma_1', \beta)
   &
   \text{(\ref{lem:aux:list-gen:gc})}
   \locallabel{listgen:ih2}
   \\
   &
   \sigma_1, p \dbwdarrow \leq e_2
   &
   \text{(\ref{lem:aux:clause:gc})}
   \notag
   \\
   &
   \sigma_1', p \dbwdarrow e_2' \leq e_2 \quad (\exists e_2')
   &
   \text{(Monotonicity)}
   \locallabel{listgen:ih3}
   \\
   &
   e_{2}, \exListComp{s_2}{\vec{q}} \dbwdarrow \leq \annot{\exListComp{s_{2}}{\vec{q}}}{\alpha}
   &
   \text{(IH)}
   \notag
   \\
   &
   e_{2}', \exListComp{s_2}{\vec{q}} \dbwdarrow \annot{\exListComp{s_{2}'}{\vec{q}'}}{\beta'} \leq \annot{\exListComp{s_2}{\vec{q}}}{\alpha}  \quad (\exists s_2', \vec{q}', \beta')
   &
   \text{(Monotonicity)}
   \locallabel{listgen:ih4}
   \\
   &
   \qedLocal
   \derivation{\derivationWidth}{
      \begin{smathpar}
         \inferrule*[
            lab={\ruleName{$\dbwdarrow$-list-comp-gen}},
         ]
         {
            e \eq \exApp{\exApp{\kw{concatMap}}{(\exLambda{\sigma_2})}}{e_1}
            \\
            e_{1}, s_{1} \dbwdarrow s_{1}'
            \\
            \totalisebwd{\sigma_2}{p}{\sigma_1'}{\beta}
            \\
            \sigma_{1}', p \dbwdarrow e_{2}'
            \\
            e_{2}', \exListComp{s_2}{\vec{q}} \dbwdarrow \annot{\exListComp{s_{2}'}{\vec{q}'}}{\beta'}
         }
         {
            e,
            \exListComp{s_{2}}{\qualGenerator{p}{s_{1}} \concat \vec{q}}
            \dbwdarrow
            \annot{\exListComp{s_{2}'}{\qualGenerator{p}{s_{1}'} \concat \vec{q}'}}{\beta \join \beta'}
         }
      \end{smathpar}
   }
   &
   \text{(\localref{listgen:ih1}, \localref{listgen:ih2}, \localref{listgen:ih3}, \localref{listgen:ih4})}
   \notag
   \\
   &
   \qedLocal
   \annot{\exListComp{s_{2}'}{\qualGenerator{p}{s_{1}'} \concat \vec{q}'}}{\beta \join \beta'} \leq \annListComp{s_{2}}{\qualGenerator{p}{s_{1}} \concat \vec{q}}{\alpha}
   &
   \text{(\localref{listgen:ih1}, \localref{listgen:ih2}, \localref{listgen:ih3}, \localref{listgen:ih4})}
   \notag
\end{flalign}
\end{proof}
