\begin{frame}[t,fragile]
\frametitle{Relationship to pattern-matching}
\vspace{10pt}
\begin{lstlisting}[escapechar=&]
zipWith $\oplus$ [] ys = []
zipWith $\oplus$ (x $\Cons$ xs) [] = []
zipWith $\oplus$ (x $\Cons$ xs) (y $\Cons$ ys) = (x $\oplus$ y) $\Cons$ (zipWith $\oplus$ xs ys)
\end{lstlisting}
\only<2>{
\[
\text{\emph{one of}\quad}
\begin{cases}
\ttt{($\Unk$, [], $\Unk$)} \\
\ttt{($\Unk$, $\Unk$ $\Cons$ $\Unk$, [])} \\
\ttt{($\Unk$, $\Unk$ $\Cons$ $\Unk$, $\Unk$ $\Cons$ $\Unk$)}
\end{cases}
\]
}
\only<3->{
\[
\text{\emph{one of}\quad}
\begin{cases}
\ttt{(\zap{\oplus}, [], \zap{\ttt{ys}})} \\
\ttt{(\zap{\oplus}, \zap{\ttt{x}} $\Cons$ \zap{\ttt{xs}}, [])} \\
\ttt{(\zap{\oplus}, \zap{\ttt{x}} $\Cons$ \zap{\ttt{xs}}, \zap{\ttt{y}} $\Cons$ \zap{\ttt{ys}})}
\end{cases}
\]
}
\onslide<4->{

  select the relevant equation.
\end{itemize}
}
