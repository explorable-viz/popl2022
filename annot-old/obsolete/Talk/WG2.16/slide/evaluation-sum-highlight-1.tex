\begin{frame}
\frametitle{Demand-indexed evaluation}
Typing rule for a demand of type $\tySum{A}{B}$:
\begin{mathpar}
\inferrule*
{
  \Gamma \vdash \Highlight{\Trie{\sigma}{W}{A}}
  \\
  \Gamma \vdash \Trie{\tau}{W}{B}
}
{
  \Gamma \vdash \Trie{\trieSum{\Highlight{\sigma}}{\tau}}{W}{(\tySum{A}{B})}
}
\end{mathpar}
Evaluation rules (intro only):
\begin{mathpar}
\inferrule*
{
  \eval{e}{\textcolor{blue}{\sigma}}{v}
}
{
  \eval{\exInl{e}}{\trieSum{\textcolor{blue}{\sigma}}{\tau}}{\exInl{v}}
}
%
\and
%
\inferrule*
{
  \eval{e}{\tau}{v}
}
{
  \eval{\exInr{e}}{\trieSum{\sigma}{\tau}}{\exInr{v}}
}
\end{mathpar}
\end{frame}
